-- Создание таблицы "группа"
CREATE TABLE группа (
    имя VARCHAR(50),
    рейтинг INTEGER,
    курс INTEGER
);

-- Вставка данных в таблицу
INSERT INTO группа (имя, рейтинг, курс)
VALUES ('Группа 1', 75, 2),
       ('Группа 2', 30, 1),
       ('Группа 3', 45, 3);

-- группы с рейтингом <= 50
SELECT имя, рейтинг, курс
FROM группа
WHERE рейтинг <= 50;
